⚙️ Configuration Supabase

Renseignez l'URL et la clé publique (anon key) de votre projet Supabase. Ces informations sont visibles dans Project Settings → API de votre tableau de bord Supabase.

🗄️ Initialiser la base de données

Copiez ce SQL dans l'éditeur SQL de Supabase (SQL Editor → New query) pour créer les tables nécessaires.

-- Clients
CREATE TABLE clients (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  nom text NOT NULL,
  type text,
  contact text,
  email text,
  telephone text,
  adresse text,
  departement text,
  notes text,
  created_at timestamptz DEFAULT now()
);

-- Contrats / abonnements
CREATE TABLE contrats (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  client_id uuid REFERENCES clients(id) ON DELETE CASCADE,
  produit text NOT NULL,
  detail text,
  periodicite text NOT NULL DEFAULT 'Annuel',
  montant_ht numeric(10,2),
  derniere_facture date,
  echeance_contractuelle date,
  facture_ebp text,
  notes text,
  actif boolean DEFAULT true,
  created_at timestamptz DEFAULT now()
);

-- Index utiles
CREATE INDEX ON contrats(client_id);
CREATE INDEX ON contrats(echeance_contractuelle);
CREATE INDEX ON contrats(produit);
